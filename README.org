Example:

#+BEGIN_SRC lisp
(in-package :arc-user)
(named-readtables:in-readtable :arc)


(map [car cdr._] '((1 2) (1 2) (1 2) (1 2) (1 2)))
;=>  (2 2 2 2 2)


(map #'list:car:cdr '((1 2) (1 2) (1 2) (1 2) (1 2)))
;=>  ((2) (2) (2) (2) (2))


(accum acc
  (each e (range 0 10)
    (when odd.e acc.e)))
;=>  (1 3 5 7 9)
#+END_SRC lisp
